router.post('/networks', ensureAuthenticated, function(req, res, next) {
  if(req.user.admin == 1){
    var street = req.body.street;
    var number = req.body.number;

    if (!errors){
      Network.getNetworkByStreetAndHousenumber(street, number, function(err, networks){
        if (err) throw err;
        res.render('panel/networkList', {
          networks: networks,
          csrfToken: req.csrfToken()
        });
      });
    }else{
      res.redirect("/");
    }
  }else{
    res.redirect("/");
  }
});
